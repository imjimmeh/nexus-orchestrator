# EPIC-205 W3+W4: Approval Dispatch, Completion Bookkeeping & Scope Confirmation UI — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire skill proposal approval to dispatch the `create_skill` workflow (W3), then let a human confirm or override the recommended scope from the UI (W4).

**Architecture:** W3 — `SkillProposalService.approve()` emits `SKILL_PROPOSAL_APPROVED_EVENT` on `EventEmitter2`; a new `SkillProposalApprovedListener` calls `WorkflowEngineService.startWorkflow('create_skill', {...})`; a new `SkillProposalCompletionListener` listens on `WORKFLOW_RUN_COMPLETED_EVENT`, detects skill-proposal runs via `source_proposal_id` in trigger data, and writes `applied`/`failed` status + `scope_confirmation` diagnostics. W4 — `confirmScope()` on `SkillProposalService` reads the SKILL.md, injects the confirmed scope into frontmatter via js-yaml, and clears the pending flag; a new `POST skills/proposals/:id/confirm-scope` endpoint; frontend renders the `applied` state, a scope-confirm card, and optional scope override.

**Tech Stack:** TypeScript, NestJS, Zod (`@nexus/core`), Vitest, js-yaml, React, TanStack Query. Test commands: `npm run test:api -- <relative-path>` (API), `npm run test --workspace=packages/core -- <relative-path>` (core). Build: `npm run build:api`. Web typecheck: `cd apps/web && npx tsc --noEmit`.

## Key Architecture Notes

**EventEmitter2 is globally injectable** — `EventEmitterModule.forRoot()` in `AppModule` makes `EventEmitter2` injectable anywhere without module-level wiring. New listeners need only be added to `LearningModule.providers`.

**`WorkflowEngineService` availability** — `LearningService` already injects `WorkflowEngineService`, so it flows through the existing `forwardRef(() => MemoryModule)` import. New listeners in `LearningModule` can inject it directly.

**Detecting skill-proposal runs in the completion listener** — `WorkflowRunEvent.workflowId` is a DB UUID, not the YAML `workflow_id` string. Avoid an extra DB lookup by checking `stateVariables.trigger.source_proposal_id` — if it's a non-empty string, this is a skill-proposal run.

**Reading job output:**
```typescript
const jobs = readRecord(event.stateVariables.jobs);
const authorJob = readRecord(jobs?.['author_skill']);
const output = readRecord(authorJob?.output);
// output: { skill_name, materialized, recommended_scope, scope_rationale, rejection_reason? }
```

**`diagnostics_json` merge** — load the proposal first, spread existing `diagnostics_json`, then add `scope_confirmation`. Do not overwrite blindly.

**`SkillScopeSchema`** is in `packages/core/src/schemas/ai-config/skills.schema.ts` (added in W1). Import it from there for `confirmSkillProposalScopeSchema`.

```typescript
// skill-proposal-approved.listener.ts
import { Injectable, Logger } from '@nestjs/common';
import { OnEvent } from '@nestjs/event-emitter';
import { WorkflowEngineService } from '../../workflow/workflow-engine.service';
import {
  SKILL_PROPOSAL_APPROVED_EVENT,
  type SkillProposalApprovedEvent,
} from './skill-proposal.events';

@Injectable()
export class SkillProposalApprovedListener {
  private readonly logger = new Logger(SkillProposalApprovedListener.name);

  constructor(private readonly workflowEngine: WorkflowEngineService) {}

  @OnEvent(SKILL_PROPOSAL_APPROVED_EVENT)
  async handleSkillProposalApproved(
    event: SkillProposalApprovedEvent,
  ): Promise<void> {
    const runId = await this.workflowEngine.startWorkflow('create_skill', {
      target_skill_name: event.targetSkillName,
      patch_markdown: event.patchMarkdown,
      proposal_summary: event.proposalSummary,
      source_proposal_id: event.proposalId,
      scope_id: event.scopeId ?? '',
    });

    if (!runId) {
      this.logger.warn(
        `Failed to start create_skill workflow for proposal ${event.proposalId}`,
      );
    } else {
      this.logger.log(
        `Started create_skill workflow run ${runId} for proposal ${event.proposalId}`,
      );
    }
  }
}
```

- [ ] **Step 2: Create the listener**

```typescript
// skill-proposal-completion.listener.ts
import { Injectable, Logger } from '@nestjs/common';
import { OnEvent } from '@nestjs/event-emitter';
import { WORKFLOW_RUN_COMPLETED_EVENT } from '../../workflow/workflow-events.constants';
import type { WorkflowRunEvent } from '../../workflow/workflow-events.types';
import { SkillImprovementProposalRepository } from '../database/repositories/skill-improvement-proposal.repository';

@Injectable()
export class SkillProposalCompletionListener {
  private readonly logger = new Logger(SkillProposalCompletionListener.name);

  constructor(
    private readonly proposals: SkillImprovementProposalRepository,
  ) {}

  @OnEvent(WORKFLOW_RUN_COMPLETED_EVENT)
  async handleWorkflowCompleted(event: WorkflowRunEvent): Promise<void> {
    const trigger = readRecord(event.stateVariables.trigger);
    const proposalId = readNonEmptyString(trigger?.source_proposal_id);
    if (!proposalId) {
      return;
    }

    const proposal = await this.proposals.findById(proposalId);
    if (!proposal) {
      this.logger.warn(
        `Proposal ${proposalId} not found for completed workflow run ${event.workflowRunId}`,
      );
      return;
    }

    const jobs = readRecord(event.stateVariables.jobs);
    const authorJob = readRecord(jobs?.['author_skill']);
    const output = readRecord(authorJob?.output);
    const materialized = output?.materialized === true;

    if (materialized) {
      const recommendedScope = readRecord(output?.recommended_scope) ?? null;
      const scopeRationale = readNonEmptyString(output?.scope_rationale) ?? null;

      await this.proposals.updateById(proposalId, {
        status: 'applied',
        applied_at: new Date(),
        diagnostics_json: {
          ...(proposal.diagnostics_json ?? {}),
          scope_confirmation: {
            pending: true,
            recommended_scope: recommendedScope,
            scope_rationale: scopeRationale,
          },
        },
      });
    } else {
      const rejectionReason =
        readNonEmptyString(output?.rejection_reason) ??
        'Skill materialization was not completed by the authoring agent';

      await this.proposals.updateById(proposalId, {
        status: 'failed',
        error_message: rejectionReason,
      });
    }
  }
}

function readRecord(value: unknown): Record<string, unknown> | null {
  if (!value || typeof value !== 'object' || Array.isArray(value)) {
    return null;
  }
  return value as Record<string, unknown>;
}

function readNonEmptyString(value: unknown): string | null {
  if (typeof value !== 'string') {
    return null;
  }
  const trimmed = value.trim();
  return trimmed.length > 0 ? trimmed : null;
}
```

```typescript
function extractScopeConfirmation(
  diagnostics: Record<string, unknown> | null,
): {
  pending: boolean;
  recommended_scope: Record<string, unknown> | null;
  scope_rationale: string | null;
  confirmed_scope?: Record<string, unknown> | null;
} | null {
  if (!diagnostics) return null;
  const sc = diagnostics.scope_confirmation;
  if (!sc || typeof sc !== 'object' || Array.isArray(sc)) return null;
  const confirmation = sc as Record<string, unknown>;
  const recommended =
    confirmation.recommended_scope &&
    typeof confirmation.recommended_scope === 'object' &&
    !Array.isArray(confirmation.recommended_scope)
      ? (confirmation.recommended_scope as Record<string, unknown>)
      : null;
  const confirmed =
    confirmation.confirmed_scope !== undefined
      ? (confirmation.confirmed_scope as Record<string, unknown> | null)
      : undefined;
  return {
    pending: confirmation.pending === true,
    recommended_scope: recommended,
    scope_rationale:
      typeof confirmation.scope_rationale === 'string'
        ? confirmation.scope_rationale
        : null,
    ...(confirmed !== undefined ? { confirmed_scope: confirmed } : {}),
  };
}
```

```typescript
// New method:
async confirmScope(
  id: string,
  dto: ConfirmSkillProposalScopeRequest,
): Promise<SkillProposalListItem> {
  const proposal = await this.proposals.findById(id);
  if (!proposal) {
    throw new NotFoundException(`Proposal ${id} not found`);
  }
  if (proposal.status !== 'applied') {
    throw new ConflictException(
      `Proposal ${id} is not in applied status (current: ${proposal.status})`,
    );
  }

  const skillRecord = this.skillLibrary.findByName(proposal.target_skill_name);
  if (skillRecord) {
    const updatedMarkdown = this.buildScopedMarkdown(
      skillRecord.skillMarkdown,
      dto.scope as Record<string, string[]> | null,
    );
    this.skillsService.updateSkill(proposal.target_skill_name, {
      skill_markdown: updatedMarkdown,
    });
  }

  const existingDiagnostics = this.asRecord(proposal.diagnostics_json ?? {});
  const existingScopeConfirmation = this.asRecord(
    existingDiagnostics?.scope_confirmation ?? {},
  );

  const updated = await this.proposals.updateById(id, {
    diagnostics_json: {
      ...existingDiagnostics,
      scope_confirmation: {
        ...existingScopeConfirmation,
        pending: false,
        confirmed_scope: dto.scope ?? null,
      },
    },
  });

  if (!updated) {
    throw new NotFoundException(`Proposal ${id} not found after update`);
  }

  return toSkillProposalListItem(updated);
}

private buildScopedMarkdown(
  currentMarkdown: string,
  scope: Record<string, string[]> | null,
): string {
  const match = currentMarkdown.match(/^(---\n)([\s\S]*?)(\n---)([\s\S]*)$/);
  if (!match) {
    return currentMarkdown;
  }
  const frontmatter = yaml.load(match[2]) as Record<string, unknown>;
  if (scope === null) {
    delete frontmatter.scope;
  } else {
    const scopeObj: Record<string, string[]> = {};
    const s = scope as { projects?: string[]; agents?: string[]; workflows?: string[] };
    if (s.projects?.length) scopeObj.projects = s.projects;
    if (s.agents?.length) scopeObj.agents = s.agents;
    if (s.workflows?.length) scopeObj.workflows = s.workflows;
    if (Object.keys(scopeObj).length > 0) {
      frontmatter.scope = scopeObj;
    } else {
      delete frontmatter.scope;
    }
  }
  const serialized = yaml.dump(frontmatter).trimEnd();
  return `---\n${serialized}\n---${match[4]}`;
}
```

```tsx
function ScopeConfirmationCard({

  confirmation,
  onConfirmScope,

  const [isOverriding, setIsOverriding] = useState(false);
  const [overrideProjects, setOverrideProjects] = useState('');
  const [overrideAgents, setOverrideAgents] = useState('');
  const [overrideWorkflows, setOverrideWorkflows] = useState('');

  if (!confirmation.pending) {

function formatScopeDisplay(scope: {
  projects?: string[];
  agents?: string[];
  workflows?: string[];
}): string {
  const parts: string[] = [];
  if (scope.projects?.length) parts.push(`projects: ${scope.projects.join(', ')}`);
  if (scope.agents?.length) parts.push(`agents: ${scope.agents.join(', ')}`);
  if (scope.workflows?.length) parts.push(`workflows: ${scope.workflows.join(', ')}`);
  return parts.length > 0 ? parts.join(' · ') : 'Global';
}

function buildScopeFromInputs(
  projects: string,
  agents: string,
  workflows: string,
): { projects?: string[]; agents?: string[]; workflows?: string[] } | null {
  const parse = (s: string) =>
    s.split(',').map((v) => v.trim()).filter((v) => v.length > 0);
  const scopeObj: { projects?: string[]; agents?: string[]; workflows?: string[] } = {};
  const p = parse(projects);
  const a = parse(agents);
  const w = parse(workflows);
  if (p.length) scopeObj.projects = p;
  if (a.length) scopeObj.agents = a;
  if (w.length) scopeObj.workflows = w;
  return Object.keys(scopeObj).length > 0 ? scopeObj : null;
}
```
